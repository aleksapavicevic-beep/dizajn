// FIRESTORE_SECURITY_RULES.txt
// Copy-paste u Firestore -> Rules tab

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow authenticated users to access their own data
    match /users/{userId} {
      allow read, write: if request.auth.uid == userId;
    }

    // Routines - dostupno samo vlasniku ili caregiver-u
    match /routines/{document=**} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == resource.data.userId);
    }

    // Reminders - dostupno samo vlasniku
    match /reminders/{document=**} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == resource.data.userId);
    }

    // Family members - dostupno samo vlasniku
    match /familyMembers/{document=**} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == resource.data.userId);
    }

    // Emergency contacts - dostupno samo vlasniku
    match /emergencyContacts/{document=**} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == resource.data.userId);
    }

    // Patient profiles - dostupno samo vlasniku
    match /patientProfiles/{userId} {
      allow read, write: if request.auth.uid == userId;
    }

    // Activity logs - dostupno samo vlasniku
    match /activityLogs/{document=**} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == resource.data.userId);
    }

    // Deny all ostalo
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
